EDA

We start by considering daily historical step count, which is stored in daily$steps.time. We also check whether there are any missing days, because we assume there are none.

s <-
    daily$steps.time %>%
    as.Date

all_days <-
    seq.Date(s[1], s[n], by = "day") %>%
    as.character
s <- as.character(s)

if (!all(s == all_days)) {
    stop("Missing days in input data!")
}
plot(daily$steps.steps, type = "l", ylab = "Daily steps")
